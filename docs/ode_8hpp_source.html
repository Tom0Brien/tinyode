92 template <typename Scalar, int n>
93 Result<Scalar, n> ode_euler(
94 const std::function<Eigen::Matrix<Scalar, n, 1>(const Scalar, const Eigen::Matrix<Scalar, n, 1>&)>&
95 ode_function,
96 const TimeSpan<Scalar>& time_span,
97 const Eigen::Matrix<Scalar, n, 1>& initial_conditions,
98 const Options<Scalar, n>& options) {
99
100 // Calculate the number of time steps
101 int num_time_steps = static_cast<int>((time_span.end - time_span.start) / options.fixed_time_step) + 1;
102
103 // Resize the matrices in the Result structure
104 Result<Scalar, n> result;
105 result.t.resize(1, num_time_steps);
106 result.x.resize(n, num_time_steps);
107
108 // Initialize the time and state variables with the initial conditions
109 Scalar t = time_span.start;
110 Eigen::Matrix<Scalar, n, 1> x = initial_conditions;
111
112 // Store the initial conditions in the Result structure
113 result.t(0) = t;
114 result.x.col(0) = x;
115
116 // Preallocate the derivative matrix
117 Eigen::Matrix<Scalar, n, 1> dxdt;
118
119 // Loop through the time steps and apply the Euler method to update the state variables
120 for (int i = 1; i < num_time_steps; ++i) {
121 // Compute the derivative
122 dxdt = ode_function(t, x);
123
124 // Update the state variable using the Euler method
125 x.noalias() += options.fixed_time_step * dxdt;
126
127 // Update the time variable
128 t += options.fixed_time_step;
129
130 // Store the computed state variables and corresponding time values in the Result structure
131 result.t(i) = t;
132 result.x.col(i) = x;
133 }
134
135 return result;
136 }

161
162 // Initialize the time and state variables with the initial conditions
163 Scalar t = time_span.start;

165
166 // Store the initial conditions in the Result structure
167 result.t(0) = t;
168 result.x.col(0) = x;
169
170 // Preallocate the derivative matrices
171 Eigen::Matrix<Scalar, n, 1> dxdt, k1, k2, k3, k4;
172
173 // Loop through the time steps and apply the RK4 method to update the state variables
174 for (int i = 1; i < num_time_steps; ++i) {
175 // Compute the derivatives using the current state variable and time value
176 dxdt = ode_function(t, x);
177
178 // Compute the four stages of the RK4 method
179 k1 = options.fixed_time_step * dxdt;
180 k2 = options.fixed_time_step * ode_function(t + 0.5 * options.fixed_time_step, x + 0.5 * k1);
181 k3 = options.fixed_time_step * ode_function(t + 0.5 * options.fixed_time_step, x + 0.5 * k2);
182 k4 = options.fixed_time_step * ode_function(t + options.fixed_time_step, x + k3);
183
184 // Update the state variable using the RK4 method
185 x.noalias() += (1.0 / 6.0) * (k1 + 2.0 * k2 + 2.0 * k3 + k4);
186
187 // Update the time variable
188 t += options.fixed_time_step;
189
190 // Store the computed state variables and corresponding time values in the Result structure
191 result.t(i) = t;
192 result.x.col(i) = x;

273 template <typename Scalar, int n>
274 Result<Scalar, n> ode_rk45(
275 const std::function<Eigen::Matrix<Scalar, n, 1>(const Scalar, const Eigen::Matrix<Scalar, n, 1>&)>&
276 ode_function,
277 const TimeSpan<Scalar>& time_span,
278 const Eigen::Matrix<Scalar, n, 1>& initial_conditions,
279 const Options<Scalar, n>& options) {
280 // Initialize time and state variables
281 Scalar t = time_span.start;
282 Eigen::Matrix<Scalar, n, 1> x = initial_conditions;
283
284 // Set initial time step
285 Scalar h = options.fixed_time_step;
286
287 // Set relative and absolute tolerances
288 Scalar rtol = options.relative_tolerance;
289 Scalar atol = options.absolute_tolerance;
290
291 // Initialize result structure
292 Result<Scalar, n> result;
293 result.t.resize(1, 1);
294 result.x.resize(n, 1);
295 result.t(0) = t;
296 result.x.col(0) = x;
297
298 // Initialize derivative matrices
299 Eigen::Matrix<Scalar, n, 1> k1, k2, k3, k4, k5, k6;
300
301 // Loop through time steps
302 while (t < time_span.end) {
303 // Compute k1-k6
304 k1 = ode_function(t, x);
305 k2 = ode_function(t + h * (1.0 / 4.0), x + h * (1.0 / 4.0) * k1);
306 k3 = ode_function(t + h * (3.0 / 8.0), x + h * (3.0 / 32.0) * k1 + h * (9.0 / 32.0) * k2);
307 k4 = ode_function(t + h * (12.0 / 13.0),
308 x + h * (1932.0 / 2197.0) * k1 - h * (7200.0 / 2197.0) * k2 + h * (7296.0 / 2197.0) * k3);
309 k5 = ode_function(
310 t + h,
311 x + h * (439.0 / 216.0) * k1 - h * 8.0 * k2 + h * (3680.0 / 513.0) * k3 - h * (845.0 / 4104.0) * k4);
312 k6 = ode_function(t + h * (1.0 / 2.0),
313 x - h * (8.0 / 27.0) * k1 + h * 2.0 * k2 - h * (3544.0 / 2565.0) * k3
314 + h * (1859.0 / 4104.0) * k4 - h * (11.0 / 40.0) * k5);
315 // Compute the 4th and 5th order approximations
316 Eigen::Matrix<Scalar, n, 1> x4 = x + h * (25.0 / 216.0) * k1 + h * (1408.0 / 2565.0) * k3
317 + h * (2197.0 / 4104.0) * k4 - h * (1.0 / 5.0) * k5;
318 Eigen::Matrix<Scalar, n, 1> x5 = x + h * (16.0 / 135.0) * k1 + h * (6656.0 / 12825.0) * k3
319 + h * (28561.0 / 56430.0) * k4 - h * (9.0 / 50.0) * k5
320 + h * (2.0 / 55.0) * k6;
321
322 // Compute the error estimate
323 Eigen::Matrix<Scalar, n, 1> error = x5 - x4;
324 Scalar error_norm = error.norm();
325
326 // Compute the scale factor for the tolerances
327 Eigen::Matrix<Scalar, n, 1> scale =
328 atol * Eigen::Matrix<Scalar, n, 1>::Ones() + rtol * (x.cwiseAbs().array().max(1).matrix());
329
330 // Check if error is within tolerance
331 if (error_norm < scale.norm()) {
332 // Update time and state variables
333 t += h;
334 x = x5;
335
336 // Store the computed time and state variables in the result structure
337 result.t.conservativeResize(result.t.rows(), result.t.cols() + 1);
338 result.t(result.t.size() - 1) = t;
339
340 result.x.conservativeResize(result.x.rows(), result.x.cols() + 1);
341 result.x.col(result.x.cols() - 1) = x;
342 }
343
344 // Adjust time step based on error estimate and tolerance
345 Scalar delta = 0.84 * std::pow(scale.norm() / error_norm, 0.25);
346 delta = std::max(delta, 0.1);
347 delta = std::min(delta, 4.0);
348 h *= delta;
349 }
350
351 return result;
352 }
